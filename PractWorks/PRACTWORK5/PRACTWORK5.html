def dfs_topological_sort(graph):
    visited = set()
    result = []

    def dfs(node):
        visited.add(node)
        for neighbor in graph.get(node, []):
            if neighbor not in visited:
                dfs(neighbor)
        result.append(node)

    for node in graph:
        if node not in visited:
            dfs(node)

    return result[::-1]

graph = {
    2: [7],
    4: [10, 12],
    7: [10],
    10: [],
    12: []
}

topo_order = dfs_topological_sort(graph)
