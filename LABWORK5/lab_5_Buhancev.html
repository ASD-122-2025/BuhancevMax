def linear_search(arr, target):
    for i in range(len(arr)):
        if arr[i] == target:
            return i
    return -1

def test_linear():
    results = []
    for n in range(100, 2001, 200):
        arr = np.arange(n)
        start = time.time()
        linear_search(arr, -1)
        end = time.time()
        results.append((n, end - start))
    return pd.DataFrame(results, columns=["n", "time"])

df_linear = test_linear()

def binary_search(arr, target):
    low, high = 0, len(arr) - 1
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1
    return -1

def test_binary():
    results = []
    for n in range(100, 2001, 200):
        arr = np.arange(n)
        start = time.time()
        binary_search(arr, -1)
        end = time.time()
        results.append((n, end - start))
    return pd.DataFrame(results, columns=["n", "time"])

df_binary = test_binary()
