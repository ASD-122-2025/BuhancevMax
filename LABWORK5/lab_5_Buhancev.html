def linear_search(arr, target):
    for i in range(len(arr)):
        if arr[i] == target:
            return i
    return -1

def test_linear():
    results = []
    for n in range(100, 2001, 200):
        arr = np.arange(n)
        start = time.time()
        linear_search(arr, -1)
        end = time.time()
        results.append((n, end - start))
    return pd.DataFrame(results, columns=["n", "time"])

df_linear = test_linear()

def binary_search(arr, target):
    low, high = 0, len(arr) - 1
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1
    return -1

def test_binary():
    results = []
    for n in range(100, 2001, 200):
        arr = np.arange(n)
        start = time.time()
        binary_search(arr, -1)
        end = time.time()
        results.append((n, end - start))
    return pd.DataFrame(results, columns=["n", "time"])

df_binary = test_binary()

5. Оцінити ефективність алгоритмів¶

Бінарний пошук працює значно швидше при великих розмірах масиву, але вимагає відсортованих даних. Лінійний пошук повільніший, але працює з будь-якими даними. Ефективніший — бінарний для n > 1000 і більше.

Контрольні питання¶

  1. Сформулювати задачу пошуку:
  • Знайти позицію елемента у структурі даних (масив, список тощо).
  1. Які критерії можна використовувати для оцінювання ефективності алгоритмів пошуку?:
  • Асимптотична складність, час виконання, вимоги до памʼяті.
  1. Що таке лінійний пошук і яка його асимптотична складність?:
  • Перевіряє кожен елемент. Складність: O(n)
  1. Що таке бінарний (двійковий) пошук і в чому полягає його головний принцип роботи?:
  • Розділяє масив навпіл на кожному кроці. Складність: O(log n)
  1. Які переваги і недоліки використання бінарного пошуку порівняно з лінійним?:
  • Переваги: швидкий, логарифмічний час. Недоліки: працює лише з відсортованими даними.
  1. Які ще алгоритми пошуку існують, окрім лінійного і бінарного?:
  • Інтерполяційний пошук, експоненційний пошук, хеш-пошук, пошук у дереві.
